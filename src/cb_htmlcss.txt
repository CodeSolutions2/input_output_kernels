<canvas id='canvasElement' width='256' height='256'></canvas>

<div id="output">

<script>
const canvasElement = document.getElementById('canvasElement');

const ctx = canvasElement.getContext('2d');

const y_bottom = canvasElement.height/2;

const angle = 0;
const amplitude = 20;

const frequency = 0.1;

const x = 0;

const xSpeed = 2;

// Call function for animation
animation();



function animation() {

  ctx.clearRect(0, 0, canvasElement.width, canvasElement.height);

  const y = y_bottom + Math.sin(angle)*amplitude;

  // const x = canvasElement.width/2;

  ctx.beginPath();

  const radius = 10;
  const startAngle = 0;
  const endAngle = Math.PI*2;
  ctx.arc(x, y, radius, startAngle, endAngle);

  ctx.fillStyle = 'green';

  ctx.fill();

  // update x position 
  x = x + xSpeed;

  // update angle for next frame
  angle = angle + frequency;
  
  // reset x-position 
  if (x > canvasElement.width) {
    x = 0;
  }

  // Animation that repeats the drawing
  requestAnimationFrame(animation);
  
} // end of function



</script>
