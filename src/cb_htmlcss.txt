function model(x) {
  
  // const w1 = tf.variable(tf.tensor([[784, 32]]));
  // const b1 = tf.variable(tf.tensor([32])); 

  const w1 = tf.variable(tf.tensor([784, 32]));


  // return x.pow(tf.scalar(2, "int32"))
  // OR
  // return x.matMul(w1).add(b1);

  return x.matMul(w1);
}
