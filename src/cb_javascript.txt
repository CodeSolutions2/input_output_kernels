const tf = require('@tensorflow/tfjs');
const batch_size = 1;
const n_a = 32;
const timesteps = 2*n_a;
const feature_num = 1;
const input = tf.input({batchShape: [batch_size, timesteps, feature_num]});
const lstm0 = tf.layers.lstm({units: n_a, activation: "relu", inputShape: [timesteps, feature_num], returnSequences: true });
const input_lstm0 = lstm0.apply(input);

console.log("input_lstm0: ", input_lstm0);
