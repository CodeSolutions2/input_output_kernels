A = np.array([[1, 3, 5, 7]])
print("A: ", A) 
print("np.shape(A): ", np.shape(A))

# Increases the array by a row
# A = A[np.newaxis, :]
# OR
A = np.expand_dims(A, axis=0)
print("A: ", A) 
print("np.shape(A): ", np.shape(A))

# Increases the array by a column
# A = A[:, np.newaxis]
# OR
A = np.expand_dims(A, axis=1)
print("A: ", A)
print("np.shape(A): ", np.shape(A))
