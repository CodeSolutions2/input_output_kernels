from sklearn.feature_extraction.text import CountVectorizer

text_per_document = ["the text processing of text is useful. Text processing is fun.",
                 "Is text processing useful. A document about text processing methods.", 
                 "a document about text processing."];

stop_words=["and", "is", "the", "of", "to"]

vectorizer = CountVectorizer(stop_words=stop_words)
X = vectorizer.fit_transform(text_per_document)
print(vectorizer.get_feature_names_out())
print("X: ", X.toarray())
